obj-m += hello_module.o
obj-m += proc_module.o
obj-m += char_device.o

KDIR := /lib/modules/$(shell uname -r)/build
PWD := $(shell pwd)

all:
	$(MAKE) -C $(KDIR) M=$(PWD) modules

clean:
	$(MAKE) -C $(KDIR) M=$(PWD) clean

install:
	sudo insmod hello_module.ko
	sudo insmod proc_module.ko
	sudo insmod char_device.ko

uninstall:
	sudo rmmod hello_module
	sudo rmmod proc_module
	sudo rmmod char_device

test_hello:
	sudo dmesg -C
	sudo insmod hello_module.ko
	dmesg | tail -5
	sudo rmmod hello_module
	dmesg | tail -5

test_hello_param:
	sudo dmesg -C
	sudo insmod hello_module.ko message="Custom greeting from Valeriya"
	dmesg | tail -5
	sudo rmmod hello_module
	dmesg | tail -5

test_proc:
	sudo dmesg -C
	sudo insmod proc_module.ko
	dmesg | tail -5
	cat /proc/my_config
	echo "New value from Valeriya" > /proc/my_config
	cat /proc/my_config
	sudo rmmod proc_module
	dmesg | tail -5

test_char:
	sudo dmesg -C
	sudo insmod char_device.ko
	dmesg | tail -5
	sudo mknod /dev/mydevice c 240 0
	sudo chmod 666 /dev/mydevice
	echo "Test message" > /dev/mydevice
	cat /dev/mydevice
	sudo rm /dev/mydevice
	sudo rmmod char_device
	dmesg | tail -5

.PHONY: all clean install uninstall test_hello test_hello_param test_proc test_char
